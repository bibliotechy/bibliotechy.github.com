Loops 



- name: Install dependencies
  apt:
    pkg={{ item }}
    state=present
  sudo: yes
  with_items:
    - apache2
    - php5
    - libapache2-mod-php5
    - mysql-server
    - mysql-client

Steady Loopin 



- name: Create required dirs for user sites
  file:
    path=/var/www/{{ item[0] }}/{{ item[1] }}
    state=directory
  with_nested:
    - [ 'chad', 'sara']
    - [ 'downloads', 'uploads', 'html']
  notify: restart apache
